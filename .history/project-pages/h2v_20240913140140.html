Smart crop - Horizontal to vertical
Track an object and crop around it. Can be used to change a video's aspect ration.
Code:
                
import cv2
import subprocess
from ultralytics import YOLO

# Load YOLOv8 model (pre-trained on COCO dataset)
model = YOLO('yolov8n.pt')  # 'n' is the nano version for CPU efficiency. Change to 's', 'm', 'l', 'x' for larger models.

# Input object description
object_type = input("Enter the object to track (e.g., car, person, cat): ").lower()

# Initialize lists to store x and y coordinates
x_coords = []
y_coords = []

# Define the video source (replace 'your_video_file.mp4' with your actual file path)
video_file_path = 'path/to/your_video_file.mp4'  # Change this to your file path
cap = cv2.VideoCapture(video_file_path)

frame_num = 0
while cap.isOpened():
    ret, frame = cap.read()
    if not ret:
        break

    # Perform object detection using YOLOv8
    results = model.predict(frame)

    # Extract bounding boxes and class names
    for result in results:
        boxes = result.boxes
        for box in boxes:
            top_left_x, top_left_y = int(box.xyxy[0][0]), int(box.xyxy[0][1])
            bottom_right_x, bottom_right_y = int(box.xyxy[0][2]), int(box.xyxy[0][3])
            label = model.names[int(box.cls[0])].lower()

            # Check if the detected object matches the input description (e.g., car, person, etc.)
            if object_type in label:
                # Draw bounding box on the frame
                cv2.rectangle(frame, (top_left_x, top_left_y), (bottom_right_x, bottom_right_y), (0, 255, 0), 2)
                cv2.putText(frame, label, (top_left_x, top_left_y - 10), cv2.FONT_HERSHEY_SIMPLEX, 0.9, (36, 255, 12), 2)

                # Store the top-left x and y coordinates
                x_coords.append(top_left_x)
                y_coords.append(top_left_y)

# Uncomment these line if you want to see a previe of the cropped bounding box
#     # Show the video feed with object detection
#     cv2.imshow('Object Detection and Tracking', frame)

#     frame_num += 1
                  
# # Release the video capture and close windows
# cap.release()
# cv2.destroyAllWindows()

# Define the output file name based on the detected object
output_file_name = f"{object_type}.txt"

# Write tracking data to a file in the required format
with open(output_file_name, "w") as tracking_file:
    # Write the crop line
    tracking_file.write("crop=604:1080:\n")
    
    # Prepare the x and y tracking data strings
    x_data = "x='"
    y_data = "y='"
    
    for i in range(len(x_coords)):
        x_data += f"eq(n,{i})*{x_coords[i]}+"
        y_data += f"eq(n,{i})*{y_coords[i]}+"

    # Remove the last '+' and close the string
    x_data = x_data.rstrip('+') + ":"+ "'\n"
    y_data = y_data.rstrip('+') + "'\n"
    
    # Write the data to the file
    tracking_file.write(x_data)
    tracking_file.write(y_data)

print(f"Tracking data written to {output_file_name}")

# Define paths for FFmpeg command
crop_values = f"/path/to/{object_type}.txt"
output_video_file_path = f"{object_type}_output.mp4"

# Execute FFmpeg command
ffmpeg_command = [
    "ffmpeg",
    "-i", video_file_path,
    "-an",
    "-filter_script", crop_values,
    "-y", output_video_file_path
]

subprocess.run(ffmpeg_command)

print(f"FFmpeg processing complete. Output video saved as {output_video_file_path}")
