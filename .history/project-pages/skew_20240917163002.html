Change frame perspective
Change perspective over time to get fake 3D rotation.
Code:
                input='path/to/input.jpg' # image input
output='path/to/output.mp4' # video output

topleft='0:(0+100-(1*in))'
topright='W:(0+100+(1*in))'
bottomleft='0:(H-100+(1*in))'
bottomright='W:(H-100-(1*in))'
duration=50 # in frames

ffmpeg \
-i $input \
-vf \
"scale=1280:720,\
loop=$duration:1:1,\
perspective=$topleft:$topright:$bottomleft:$bottomright:sense=source:eval=frame" \
-y $output
