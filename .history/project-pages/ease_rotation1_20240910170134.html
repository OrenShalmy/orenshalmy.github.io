Animated Rotation with easing
Rotate a video from an angle to another angle, over time with easing in/out.
Code:
                output='/path/to/output.mp4'
start_angle=0
end_angle=-90
anim_duration=2
easing_factor=-5

ffmpeg -loop 1 -i $input -t $anim_duration -filter_complex \
"[0:v]rotate=angle=(${start_angle} * PI/180) - (((1 - exp(${easing_factor} * (t / ${anim_duration}))) / (1 - exp(${easing_factor}))) * (${start_angle} - ${end_angle}) * PI/180)[rotated]" \
-map "[rotated]" -y $output
