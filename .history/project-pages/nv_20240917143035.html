Retro Look
Use this combination of filters to achieve a retro style look. Somewhat inspired by this work.
Code:
                input='/Users/oren/Downloads/2024-09-11-LC-CO-OFRV-00014-011-scaled.mp4'
output='/Users/oren/Downloads/nightvision.mp4'

ffmpeg \
-i $input \
-filter_complex \
"[0]drawgrid=w=(iw+10)/2:h=(ih+10)/2:t=2:c=white@0.5[cross];\
[cross]curves=preset=vintage[look];\
[look]colorize=hue=100[hue];\
[hue]eq=gamma=1.7:contrast=1.4[eq];\
[eq]vignette='PI/3'[vignette];\
[vignette]noise=alls=50:allf=t+u[noise];\
[noise]gblur=1[out]" \
-map "[out]" -an -y $output
